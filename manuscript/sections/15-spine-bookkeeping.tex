\section{Spine bookkeeping: margins, small primes, depth, assembly}
\label{sec:bookkeeping}

\subsection{A binomial digit-counting sublemma}

\begin{lemma}[few carries are rare]\label{lem:fewcarries}
\statusP\quad
Fix a prime $p$ and integers $0\le e<J$, and set $D=J-e$. Among $n\le x$
in any fixed congruence class modulo $p^{e}$, the proportion whose
base-$p$ digit data at positions $e<j\le J$ produce fewer than $D/8$
carries is at most $e^{-D/50}+O(p^{J}/x)$.
\end{lemma}

\begin{proof}
The digit vector $(n\bmod p^{J})$ equidistributes over admissible
classes up to $O(p^{J}/x)$ each. Condition bottom-up: given all digits
below position $j$, the carry event $\fr{n/p^{j}}\ge\tfrac12$ holds for
at least $(p-1)/2$ of the $p$ values of the digit $d_{j-1}$
(unconditionally for $d_{j-1}\ge\tfrac{p+1}2$; the boundary digit
depends on the lower history), so the conditional carry probability is
$\ge\tfrac{p-1}{2p}\ge\tfrac13$ at every position, uniformly in the
history ($=\tfrac12$ for $p=2$). The carry count therefore
stochastically dominates $\mathrm{Bin}(D,\tfrac13)$, and by Chernoff
$\Pr[\mathrm{Bin}(D,\tfrac13)<D/8]\le e^{-D/50}$.
\end{proof}

\begin{lemma}[margin successes are plentiful]\label{lem:marginwins}
\statusP\quad
For each prime $p$ there is a two-digit pattern $(a,b)$ such that
$d_{j-1}=a$, $d_{j-2}=b$ forces $\fr{m/p^{j}}\in
I^{(+)}_j=[\tfrac12,1-p^{-j})$, for every $j\ge3$. Consequently, among
$m\le x$ in a fixed class mod $p^{e}$, the proportion with fewer than
$D'/(4p^{2})$ margin-strict successes at the positions
$j=e+3,e+5,\ldots$ (with $D'=\lfloor(J-e-2)/2\rfloor$ disjoint digit
pairs) is at most $e^{-cD'/p^{2}}+O(p^{J}/x)$, $c$ absolute.
\end{lemma}

\begin{proof}
For $p=2$ take $(a,b)=(1,0)$: then
$\fr{m/2^{j}}\in[\tfrac12,\tfrac12+\tfrac14)\subset I^{(+)}_j$ for
$j\ge2$. For odd $p$ take $(a,b)=(\tfrac{p+1}2,0)$: then
$\fr{m/p^{j}}\in[\tfrac{p+1}{2p},\tfrac{p+1}{2p}+p^{-2})\subset
I^{(+)}_j$ for $j\ge3$. The patterns sit on disjoint digit pairs, so
the success events are independent uniform draws with probability
$p^{-2}$ each; Chernoff applies.
\end{proof}

\subsection{Lemma 0$'$: margins are free (full proof)}

\begin{proof}[Proof of Lemma~\ref{lem:margins}]
It suffices to show $\dens(\Dm\setminus\Dp)=0$. Let
$m\in\Dm\setminus\Dp$: for some $p^{e}\,\|\,m$,
$\ka^{(+)}_p(m)<e\le\ka^{(-)}_p(m)$, forcing (i) a digit in a margin
window at some level, and (ii) fewer than $e$ margin-strict successes.

Fix $\varepsilon>0$, split at $P_0(\varepsilon)$.
\emph{Large $p>P_0$:} condition (i): the margin windows at level $j$
are at most $4$ classes mod $p^{j}$, impossible at $j=1$ jointly with
$p\mid m$, density $\le4p^{-j}$ for $j\ge2$; total
$\sum_{p>P_0}8p^{-2}<\varepsilon$.
\emph{Small $p\le P_0$:} fix $p$ and $e=v_p(m)$, take
$J=\lfloor\log_p\sqrt x\rfloor$, $D'=\lfloor(J-e-2)/2\rfloor$. If
$e\le D'/(4p^{2})$, condition (ii) puts $m$ in the exceptional set of
Lemma~\ref{lem:marginwins}: density $\le
p^{-e}(e^{-cD'/p^{2}}+O(x^{-1/2}))$, and $\sum_{e}p^{-e}e^{-cD'/p^2}
=o_x(1)$ for fixed $p$ since $D'\to\infty$. If $e>D'/(4p^{2})$, then
$p^{-e}\le\exp(-cJ\log p/(8p^{2}))=x^{-c'(p)}$, again $o_x(1)$ summed
over $e\le2J$ (and $e>2J$ has density $\le p^{-2J}\le x^{-1}$).
Hence $\limsup_x\dens_x(\Dm\setminus\Dp)\le\varepsilon$ for every
$\varepsilon$.
\end{proof}

\noindent The status of Lemma~\ref{lem:margins} is hereby upgraded to
\statusP.

\subsection{Lemma A: proof}

\begin{proof}[Proof of Lemma~\ref{lem:A}]
Fix $p\le x^{\delta}$ and $v\ge1$, and count $n\le x$ with
$p^{v}\mid n$ but $\ka_p(n)<v$. Take $J_p=\lfloor\log_p\sqrt
x\rfloor$ and $D=J_p-v$ (failure overall implies failure among
positions $\le J_p$, the correct monotone direction; the
equidistribution error is then $O(x^{-1/2})$ per class). If
$v\le D/8$: by Lemma~\ref{lem:fewcarries} the count is $\ll
xp^{-v}(e^{-D/50}+x^{-1/2})$. If $v>D/8$: then
$p^{v}\ge p^{D/8}\ge(\sqrt x\,p^{-v})^{1/8}$, so $p^{v}\ge x^{1/18}$
and the trivial count $x/p^{v}\le x^{1-1/18}$ suffices, summed over
the $O(\log^2x)$ such pairs $(p,v)$. For the main range, sum over
$p$ by dyadic exponent: for $p\in(x^{\delta/2^{k+1}},x^{\delta/2^{k}}]$
one has $D\ge2^{k}/(4\delta)$ (for $v\le D/8$), Mertens gives
$\sum1/p=\log2+o(1)$ per range, and
$\sum_{k\ge0}e^{-c2^{k}/\delta}\ll e^{-c/\delta}$ geometrically ---
no $\log\log$ loss. The $v$-sum is geometric in $p^{-v}$. The shifted
side is identical with the low digits $(1,0^{v-1})$ (carry-free for
$p$ odd; for $p=2$, $j=1$ the digit gives a free carry --- the
favorable direction). The square-discard costs
$\sum_{p>x^{\delta}}x/p^{2}\ll x^{1-\delta}$.
\end{proof}

\noindent Lemma~\ref{lem:A} is hereby \statusP{} (the counting input
being Lemma~\ref{lem:fewcarries}; the statement matches
Ford--Konyagin's \S2 framework, cited for provenance).

\subsection{Depth packaging (ledger D5)}

\begin{proposition}[higher exponents and deeper slots]\label{prop:depth}
\statusP{} (statements; proofs are verbatim isomorphic to \S\ref{sec:dls})
\quad
Let $e\ge1$ and consider $q^{e}\,\|\,n-1$ with binding slots
$e<j\le J_q$. (i) The slot-$j$ window indicator is a function of
$n\bmod q^{j}$ with exact finite Fourier expansion whose coefficients
decay like $\langle\lambda\rangle^{-1}$. (ii) The depth-$d$ character
family ($d=j-e$) --- characters of $(\Z/q^{j})^*$ trivial on the
index-$q^{d}$ subgroup --- obeys the analogues of
Propositions~\ref{prop:fiber},~\ref{prop:energy},
Theorem~\ref{thm:dls} and Corollary~\ref{cor:ap} with $q^{2}$ replaced
by $q^{j}$ and the same proofs (the kernel filtration
$1+q^{j-1}\Z/q^{j}$ playing the role of $1+q\Z/q^{2}$; the
progression-restricted version is again the classical sieve in the
progression coordinate). (iii) Slot-boundary slivers (sizes with
$q^{j}\in(x^{1-\varepsilon'},2x]$ near a boundary) have log-measure
$O(\varepsilon')$ and join the trim registry. (iv) Multi-prime
configurations multiply the character data by CRT, with conductors
$\prod q_i^{j_i}$, and the bilinearity
$\fq(ab)=\fq(a)+\fq(b)$ persists at every depth.
\end{proposition}

\subsection{The assembly cascade (ledger S3)}

\begin{proposition}[composition of A--D]\label{prop:cascade}
\statusP{} given Lemmas A, B, C, D in their stated forms\quad
Assume: Lemma~\ref{lem:A} with parameter $\delta$; Lemma~\ref{lem:B}
with rate $\Llog^{-c_B}$ at almost all scales;
Lemma~\ref{lem:C} with rate $\Llog^{-c_C}$; Estimate~D$^\dagger$
(equivalently Lemma~\ref{lem:D} via
\S\S\ref{sec:reduction}--\ref{sec:major}) with rate $\Llog^{-c_D}$ on
band average; Lemma~\ref{lem:margins}; and the trim registry with
total cost $O(\eta_{\mathrm{tot}})$. Then, at almost all scales,
\[
\frac1{\log x}\sum_{n\le x}\frac{\mathbf 1_{\W}(n)}{n}
=c_1^{2}+O\big(\Llog^{-c}\big)+O\big(\varepsilon(\delta)\big)
+O(\eta_{\mathrm{tot}}),\qquad c=c(c_B,c_C,c_D,K)>0,
\]
and Theorem~\ref{thm:main} follows by $\delta\to0$,
$\eta_{\mathrm{tot}}\to0$ after $x\to\infty$.
\end{proposition}

\begin{proof}
By Lemma~\ref{lem:sandwich} and Lemma~\ref{lem:margins} it suffices to
evaluate the correlation for $\D\cap(D_0^{(\pm)}+1)$. Remove the
Lemma-A exceptional set ($O(\varepsilon(\delta))$ density) and the
trims. Partition the remaining $n$ by anatomy-class pairs at a fixed
finite resolution $\rho$ (boxes for both $n$ and $n-1$); the number of
classes is $O_\rho(1)$. Within each class pair, write the joint
indicator as (anatomy pair indicator)$\times$(digit indicator of $n$
given class)$\times$(digit indicator of $n-1$ given the rest). Replace
the third factor by its class mean via Lemma~\ref{lem:D} (cost
$\Llog^{-c_D}$ per class, times trivially bounded remaining factors);
replace the second by its class mean via Lemma~\ref{lem:C}, the
remaining $(n-1)$-side data (anatomy only, after the D-step) being
rendered $1$-bounded \emph{multiplicative} by the generalized
interpolation of Lemma~\ref{lem:interp} (general bands above
$x^{\delta}$, bounded multiplicity $\le1/\delta$, $O_{\delta}(1)$
nodes --- ledger item W6) (cost $\Llog^{-c_C}$); replace the anatomy pair probability by the product
via Lemma~\ref{lem:B} (cost $\Llog^{-c_B}$). Each step's exceptional
scale set has logarithmic density $\ll\Llog^{-c}$; their union over
the $O_\rho(1)$ classes likewise. Summing the factored class
contributions reconstitutes
$\big(\sum_{\text{classes}}P[\mathcal A]\,P[\mathcal R\mid\mathcal
A]\big)^{2}$ up to resolution error $O(\rho)$, and the inner sum is
Ford--Konyagin's density computation \cite{FK21}, $=c_1+o(1)$.
Choosing $\rho=\rho(x)\to0$ slowly and folding $O(\rho)$ into the
error completes the proof.
\end{proof}
