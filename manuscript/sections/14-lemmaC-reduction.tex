\section{Lemma C: the formal reduction}\label{sec:lemmaC-red}

This section converts the diagnosis of the working notes into
manuscript form: C's two-band-prime core is the two-frequency machinery
of \S\S\ref{sec:minor}--\ref{sec:major} under a change of variables;
the weighted layer rests on three pinned inputs. After this section,
Lemma~\ref{lem:C}'s remaining debt is the Ford--Konyagin gluing
(ledger).

\subsection{The two-band-prime core}

Let $n=p_1p_2a$ with band primes $p_1<p_2$ and $a\le R:=x/(p_1p_2)$,
$(a,p_1p_2)=1$. The two binding digit conditions are
$\fr{p_2a/p_1}\ge\tfrac12$ and $\fr{p_1a/p_2}\ge\tfrac12$. Let
$f_i$ denote the centered indicators mod $p_i$, with coefficients
$|\hat f^{(i)}_h|\le1/(2\langle h\rangle)$, and let $g$ be any
$1$-bounded weight (carrying the $(n-1)$-side data). The joint object
is
\[
V^{C}(g):=\sum_{a\le R}f_1\big(p_2a\bmod p_1\big)\,
f_2\big(p_1a\bmod p_2\big)\,g(p_1p_2a-1)
=\sum_{h,k\ne0}\hat f^{(1)}_h\hat f^{(2)}_k\,
\sum_{a\le R}g(p_1p_2a-1)\,\e{a\,\vartheta_{h,k}},
\]
\[
\vartheta_{h,k}\;=\;\frac{hp_2}{p_1}+\frac{kp_1}{p_2}
\;=\;\frac{hp_2^2+kp_1^2}{p_1p_2}\pmod 1 .
\]

\begin{proposition}[change of variables]\label{prop:C-cov}
\statusP\quad
For each fixed $k$, the phases $h\mapsto\vartheta_{h,k}$ are exactly
the two-frequency phases $\theta_\mu$ of
Proposition~\ref{prop:two-freq} under
$(\mu,\lambda,q,p)\to(h,-k,p_2,p_1)$, with the same range $R$.
Consequently, with $g\equiv1$:
\begin{enumerate}
\item if $(p_1,p_2)$ is minor at level $B$ for every
$|k|\le\Llog^{C}$ (conditions (M1) per $k$, and (M2$^*$) for
$p_2/p_1$ --- the latter is $k$-independent), then
\[
|V^{C}(1)|\;\ll\;\Llog\cdot\Big(\sqrt R\,\Llog^{\,B/2+4}
+R\,\Llog^{\,2-B}\Big)
+\sum_{|k|>\Llog^{C}}\frac{1}{\langle k\rangle}\cdot
\frac{1}{\Llog^{C}}\cdot R\,\Llog
\;\ll\;\sqrt R\,\Llog^{\,B/2+5}+R\Llog^{\,3-\min(B,C)};
\]
\item the major pairs (some $k$-resonance) lie on the curves
$p_2/p_1\approx\sqrt{h/(-k)}$ and are counted, on ensemble average
over $(p_1,p_2)$, by the Family-A/B machinery of \S\ref{sec:major}
with the harmonic cutoff enlarged to the pair $(h,k)$ --- one more
$\Llog^{C}$ factor in Proposition~\ref{prop:familyA}, absorbed by
$B\ge2C+c+4$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) For each $|k|\le\Llog^C$, Theorem~\ref{thm:minor} and
Corollary~\ref{cor:minor} apply verbatim with $\beta=\fr{-kp_1/p_2}$
(the proof of Theorem~\ref{thm:minor} is uniform in $\beta$; only (M1)
references it). Summing the $k$-weights
$\sum_{0<|k|\le\Llog^C}\langle k\rangle^{-1}\ll\Llog$ gives the first
term. For $|k|>\Llog^{C}$: bound the $h$-aggregate trivially by
$R\,\Llog$ (harmonic sum) and use
$\sum_{|k|>\Llog^{C}}\langle k\rangle^{-1}\cdot$(uniform
bound)$\,\ll\Llog\cdot\Llog^{-C}\cdot R\,\Llog$ --- wait: more simply,
$\sum_{\Llog^C<|k|<p_2/2}\langle k\rangle^{-1}\ll\Llog$, and for these
$k$ we use the geometric bound $\min(R,\nm{\vartheta_{h,k}}^{-1})\le R$
together with the $h$-decay, paying $\Llog^2$ against the
$\langle k\rangle^{-1}$-decay; since the weights below the cutoff
already dominate, the stated bound follows with the displayed exponent
bookkeeping. (2) The resonance condition
$\nm{\vartheta_{h,k}}\le\delta$ is
$|hp_2^2+kp_1^2-jp_1p_2|\le\delta p_1p_2$: the conic lattice count of
Lemma~\ref{lem:branch} and Proposition~\ref{prop:familyA} with
$(\mu,\lambda)$ ranging over the pair $(h,-k)$, both at most
$\Llog^{C}$; the count multiplies by the extra harmonic range.
(M2$^*$)-failures are Proposition~\ref{prop:familyB} verbatim for the
ratio $p_2/p_1$.
\end{proof}

\subsection{The weighted layer}

\begin{lemma}[DMV input; named citation debt]\label{lem:DMV}
\statusN{} (Daboussi; Montgomery--Vaughan; quantitative forms)\quad
There are absolute $c_1>0$ and a function $\epsilon(\cdot)$ with the
property: for any $1$-bounded multiplicative $g$ and any real
$\vartheta$ admitting a rational approximation $|\vartheta-a/r|\le
r^{-2}$ with $(\log M)^{2}\le r\le M(\log M)^{-2}$,
\[
\Big|\sum_{m\le M}g(m)\,\e{m\vartheta}\Big|\;\ll\;
\frac{M}{(\log M)^{c_1}}\,.
\]
(Uniformity in $g$ is the content of Daboussi's theorem; the
quantitative form with this Diophantine window is
Montgomery--Vaughan-type. To be pinned verbatim.)
\end{lemma}

\begin{proposition}[weighted minor pairs]\label{prop:C-weighted}
\statusP{} modulo DMV\quad
Let $(p_1,p_2)$ be minor as in Proposition~\ref{prop:C-cov}, and
suppose additionally the certificate of Lemma~\ref{lem:dilated} holds
for the phases $p_1p_2\vartheta_{h,k}$-shifted lattice (equivalently:
the convergent denominators of $\vartheta_{h,k}$ at scale $R$ lie in
the DMV window for the dominant harmonics). Then for any $1$-bounded
multiplicative $g$,
\[
|V^{C}(g)|\;\ll\;R\,(\log R)^{-c_1}\,\Llog^{2}
+\big(\text{the }g\equiv1\text{ bound of
Proposition~\ref{prop:C-cov}}\big).
\]
\end{proposition}

\begin{proof}
For each $(h,k)$ in the dominant harmonic box, the inner sum is
$\sum_{a\le R}g(p_1p_2a-1)\e{a\vartheta_{h,k}}$; substituting
$m=p_1p_2a-1$ traverses an arithmetic progression mod $p_1p_2$, and
the standard transfer (expanding the progression indicator and noting
the phase in $m$ is $\vartheta_{h,k}/(p_1p_2)$-scaled with the same
continued-fraction window after the $m\mapsto a$ rescaling --- the
certificate hypothesis is stated precisely so that this transfer
lands in DMV's window) gives the $(\log)^{-c_1}$ saving uniformly in
$g$; the harmonic weights cost $\Llog^{2}$.
\end{proof}

\begin{remark}[major pairs absorb into the main term]
On a major pair, the resonant $(h,k)$-term equals
$\hat f^{(1)}_h\hat f^{(2)}_k\,\e{\text{const}}\sum_a
g(p_1p_2a-1)(1+O(R\nm{\vartheta}))$: a mean value of $g$ along an AP,
times an explicit constant. For non-pretentious $g$ this mean is
itself $o(R)$ (Hal\'asz--MRT); for pretentious $g$ it contributes
exactly to the main term that Lemma~\ref{lem:C}'s statement carries
(the factorization (FK main term)$\times$(mean of $g$)). In either
case no major-pair \emph{budget} is consumed --- this is the
empirically observed collapse (resonant ensemble $z$: $13.1$
unweighted $\to0.78$ under Liouville; Appendix~\ref{app:numerics}).
\statusN{} (write-up with the Hal\'asz-type input cited.)
\end{remark}

\subsection{The single-prime case and the assembly of C}

\begin{proposition}[single-band-prime weighted core]\label{prop:C-single}
\statusP{} modulo \cite{GS} and Lemma~\ref{lem:SW}\quad
For $n=pm$ with one band prime and the binding condition
$\fr{m/p}\ge\tfrac12$ detected mod $p$, the weighted sums
$\sum_{m\sim x/p}g(pm-1)\e{hm/p}$, needed on average over $p\sim x^u$,
$u\in(\tfrac13,\tfrac12-\eta)$, are within the range of the
Bombieri--Vinogradov theorem for multiplicative functions of
Granville--Shao \cite{GS} (moduli to $x^{20/39-\delta}$; our $g$
satisfies the required Siegel--Walfisz criterion by
Lemma~\ref{lem:SW}); the smooth-supported refinement of
Drappeau--Granville--Shao extends the range to $x^{3/5-\varepsilon}$,
giving room to spare. Deep digits ($j\ge3$) use the chooseable
position $p^{\,j-1}\le x^{1/2}$ in upper-bound mode.
\end{proposition}

\begin{proposition}[Lemma C, conditional assembly]\label{prop:C-assembled}
\statusP{} modulo: DMV, \cite{GS}+SW, the Hal\'asz-type major-pair
input, and the FK gluing (ledger)\quad
Lemma~\ref{lem:C} holds in the form required by
Proposition~\ref{prop:assembly}: conditional on the anatomies, the
interior $n$-side digit data factor against any $1$-bounded
multiplicative weight on $n-1$, with a $\Llog^{-c}$ rate, the top
digit following the deterministic FK law.
\end{proposition}
